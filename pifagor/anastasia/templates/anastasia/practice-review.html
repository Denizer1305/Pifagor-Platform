Лабораторная работа #3: Функции в Python

Сдана: 12.12.2023

Дедлайн: 15.12.2023

solution.py

def calculate_average(numbers):
# Вычисление среднего значения списка чисел
total = 0
for num in numbers:
total += num
average = total / len(numbers)
return average
def find_max(numbers):
# Поиск максимального значения в списке
max_value = numbers[0]
for num in numbers:
if num > max_value:
max_value = num
return max_value
def factorial(n):
# Вычисление факториала числа
if n == 0:
return 0 # Ошибка: факториал 0 равен 1
result = 1
for i in range(1, n+1):
result *= i
return result
# Пример использования функций
numbers = [5, 2, 8, 1, 9]
print("Среднее:", calculate_average(numbers))
print("Максимум:", find_max(numbers))
print("Факториал 5:", factorial(5))

Обработка ошибок: Рекомендуется добавить проверку на пустой список в функциях calculate_average и find_max.
